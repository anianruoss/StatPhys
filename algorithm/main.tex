\begin{frame}{Study-Algorithm Anian \visible<4->{(Brute-Force)}}

        \begin{algorithm}[H]
            \KwIn{Old bond vectors $d$, unconstrained bond vectors $d'$} 
            \KwOut{constrained bond vectors $d'$}
            \vspace{0.2cm}

			\While{$\frac{|d-d'|}{|d|}>$ tolerance}{
				\For{each bond vector $i$}{

					$f^c_i \leftarrow \frac{\mu}{2 (\Delta t)^2} \frac{d^2-d'^2}{d' \cdot d} d$ \\
					$\Delta r_{i,0} \leftarrow +\frac{(\Delta t)^2}{m_i} f^c_i$\\
					$\Delta r_{i,1} \leftarrow -\frac{(\Delta t)^2}{m_i} f^c_i$\\
					$d'_i \leftarrow d'_i + \Delta r_i$

					
				}
		}
        \end{algorithm}
        \vspace{1cm} 

        Running Time: $\Theta(June + July + August)$
